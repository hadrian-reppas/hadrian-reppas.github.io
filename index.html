Chaniel and I were most interested in the creation and acquisition date of each piece. We parsed the Creation Date and Accession Number fields to get the data:

import pandas as pd
df = pd.read_csv('AGAA.csv', sep=',', engine='python')

# simple function for parsing dates
def get_year(s):
    import re
    if not isinstance(s, str):
        return None
    finds = re.findall(r'\d\d\d\d', s)
    if len(finds) > 0:
        return int(finds[0])

# parse dates into a new dataframe
cr_dates = [get_year(c) for c, a in zip(df['Creation Date'], df['Accession Number'])
            if get_year(c) and get_year(a)]
aq_dates = [get_year(a) for c, a in zip(df['Creation Date'], df['Accession Number'])
            if get_year(c) and get_year(a)]
dates_df = pd.DataFrame({'created': cr_dates, 'acquired': aq_dates})

We wrote a function that writes the data into a .obj file:

def save_obj(file_name, arr, z_scale=1, pedestal_height=0,x0=0, y0=0, xy_scale=1):
    height, width = arr.shape
    with open(file_name, 'w') as out:
        for i in range(height):
            for j in range(width):
                x = x0 + xy_scale*j
                y = y0 + xy_scale*(height - i - 1)
                z = arr[i][j]*z_scale + pedestal_height
                out.write(f'v {x} {y} {z}')
        max_x = x0 + xy_scale*(width - 1)
        max_y = y0 + xy_scale*(height - 1)
        out.write('v 0 0 0')
        out.write(f'v {max_x} 0 0')
        out.write(f'v 0 {max_y} 0')
        out.write(f'v {max_x} {max_y} 0')
        a, b, c, d = 1, width, width*(height - 1) + 1, width*height
        e, f, g, h = d + 1, d + 2, d + 3, d + 4
        for i in range(height - 1):
            for j in range(width - 1):
                r = width*i + j + 1
                u = width*(i + 1) + j + 1
                v = width*i + (j + 1) + 1
                w = 200*(i + 1) + (j + 1) + 1
                out.write(f'f {r} {u} {v}')
                out.write(f'f {v} {u} {w}')
        if pedestal_height > 0:
            out.write(f'f {e} {g} {f}')
            out.write(f'f {f} {g} {h}')
            out.write(f'f {a} {e} {c}')
            out.write(f'f {a} {g} {e}')
            out.write(f'f {a} {b} {g}')
            out.write(f'f {h} {g} {b}')
            out.write(f'f {f} {h} {b}')
            out.write(f'f {f} {b} {d}')
            out.write(f'f {c} {f} {d}')
            out.write(f'f {c} {e} {f}')
        else:
            out.write(f'f {a} {b} {c}')
            out.write(f'f {b} {d} {c}')
